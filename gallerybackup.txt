// Lightbox — only opens if frame has a real <img> (not placeholder)
function openLb(frame){
  var img = frame.querySelector('img:not(.ph *)');
  if(!img) return; // placeholder — no lightbox
  document.getElementById('lb-img').src = img.src;
  document.getElementById('lb').classList.add('open');
}
document.getElementById('lb').addEventListener('click', function(e){
  if(e.target === this) this.classList.remove('open');
});
document.addEventListener('keydown', function(e){
  if(e.key==='Escape') document.getElementById('lb').classList.remove('open');
});

// Page exit
document.querySelectorAll('a[href]').forEach(function(a){
  var h=a.getAttribute('href');
  if(!h||h[0]==='#'||h.startsWith('http')||h.startsWith('mailto'))return;
  a.addEventListener('click',function(e){e.preventDefault();setTimeout(function(){window.location=h;},200);});
});
